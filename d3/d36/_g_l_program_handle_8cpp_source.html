94 void GLProgramHandle::SetAttributesList(std::map<std::string, std::string>& vert,
95  std::map<std::string, std::string>& frag) {
96  GLint count = 0;
97  GLsizei length = 0;
98  GLint size = 0;
99  GLenum type;
100 
101  glGetProgramiv(Program, GL_ACTIVE_ATTRIBUTES, &count);
102 
103  for (int i = 0; i < count; i++) {
104  GLchar name[32];
105  glGetActiveAttrib(Program, (GLuint) i, 32, &length, &size, &type, name);
106 
107  std::string name_str = name;
108  std::string imp_name = getImplementName(vert, name_str);
109  bool isTargetChanged = false;
110 
111  if (imp_name.empty()) {
112  imp_name = getImplementName(frag, name_str);
113  if (imp_name.empty()) continue;
114 
115  isTargetChanged = true;
116  }
117 
118  int id = glGetAttribLocation(Program, (isTargetChanged ? frag : vert).find(imp_name)->second.c_str());
119 
120  Element* element = new Element;
121  element->type = type;
122  element->id = id;
123 
124  AttributesList[imp_name] = element;
125  }
126 }
