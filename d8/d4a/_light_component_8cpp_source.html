116 void LightComponent::SetDepthMap() {
117  if(m_frameBuffer != nullptr) return;
118 
119  auto lightMgr = CORE->GetCore(LightMgr);
120  m_frameBuffer = new SFrameBuffer();
121  // 1. Generate a framebuffer.
122  // 큐브 텍스쳐도 호환되도록 수정이 필요함 (for point shadow)
123  m_frameBuffer->GenerateFramebuffer(SFrameBuffer::PLANE);
124  // 2. Generate a depth texture.
125  m_depthTexture = m_frameBuffer->GenerateTexturebuffer(SFrameBuffer::DEPTH,
126  lightMgr->SHADOW_WIDTH, lightMgr->SHADOW_HEIGHT,
127  GL_DEPTH_COMPONENT);
128  m_depthTexture->SetParameteri(GL_TEXTURE_MIN_FILTER, GL_NEAREST);
129  m_depthTexture->SetParameteri(GL_TEXTURE_MAG_FILTER, GL_NEAREST);
130 #ifdef ANDROID
131  m_depthTexture->SetParameteri(GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
132  m_depthTexture->SetParameteri(GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
133 #else
134  m_depthTexture->SetParameteri(GL_TEXTURE_WRAP_S, GL_CLAMP_TO_BORDER);
135  m_depthTexture->SetParameteri(GL_TEXTURE_WRAP_T, GL_CLAMP_TO_BORDER);
136  float borderColor[] = { 1.0, 1.0, 1.0, 1.0 };
137  m_depthTexture->SetParameterfv(GL_TEXTURE_BORDER_COLOR, borderColor);
138 #endif
139  // 3. Rasterize the framebuffer.
140  m_frameBuffer->RasterizeFramebuffer();
141 }
