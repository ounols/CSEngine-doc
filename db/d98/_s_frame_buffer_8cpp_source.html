48 void SFrameBuffer::Init(const AssetMgr::AssetReference* asset) {
49  const XNode* root = nullptr;
50 
51  try {
52  root = XFILE(asset->path.c_str()).getRoot();
53  }
54  catch (int e) {
55  SAFE_DELETE(root);
56  return;
57  }
58 
59  try {
60  const XNode& cse_framebuffer = root->getChild("CSEFRAMEBUFFER");
61  const XNode& info = cse_framebuffer.getChild("info");
62 
63  auto width_str = info.getAttribute("width").value;
64  auto height_str = info.getAttribute("height").value;
65  auto dimension_str = info.getAttribute("dimension").value;
66 
67  m_width = std::stoi(width_str);
68  m_height = std::stoi(height_str);
69  auto dimension = static_cast<BufferDimension>(std::stoi(dimension_str));
70  GenerateFramebuffer(dimension, m_width, m_height);
71 
72  const XNode& buffers = cse_framebuffer.getChild("buffers");
73  int index = 0;
74  for (const auto& buffer_node : buffers.children) {
75  auto type_str = buffer_node.getAttribute("type").value;
76  auto format_str = buffer_node.getAttribute("format").value;
77  auto isTexture_str = buffer_node.getAttribute("isTexture").value;
78 
79  auto type = static_cast<BufferType>(std::stoi(type_str));
80  auto format = std::stoi(format_str);
81  auto isTexture = std::stoi(isTexture_str) == 1;
82 
83  if(isTexture) {
84  const auto& texture = GenerateTexturebuffer(type, format);
85  const auto& textureName = "?Texture" + std::to_string(index);
86  texture->SetName(GetName() + textureName);
87  texture->SetID(GetID() + textureName);
88  }
89  else {
90  GenerateRenderbuffer(type, format);
91  }
92  ++index;
93  }
94  RasterizeFramebuffer();
95  }
96  catch (int e) {
97  SAFE_DELETE(root);
98  auto resMgr = CORE->GetCore(ResMgr);
99  resMgr->Remove(this);
100  }
101  SAFE_DELETE(root);
102 }

113 unsigned int SFrameBuffer::GenerateRenderbuffer(BufferType type, int internalFormat) {
114  if(m_dimension == CUBE) return 0;
115 
116  auto buffer = new BufferObject();
117  buffer->type = type;
118  buffer->format = internalFormat;
119 
120  glGenRenderbuffers(1, &buffer->renderbufferId);
121  glBindRenderbuffer(GL_RENDERBUFFER, buffer->renderbufferId);
122  glRenderbufferStorage(GL_RENDERBUFFER, GenerateInternalFormat(internalFormat), m_width, m_height);
123 
124  glBindFramebuffer(GL_FRAMEBUFFER, m_fbo);
125  glFramebufferRenderbuffer(GL_FRAMEBUFFER, GenerateAttachmentType(type, false), GL_RENDERBUFFER, buffer->renderbufferId);
126 
127  if(type == RENDER) {
128  if(m_mainColorBuffer == nullptr)
129  m_mainColorBuffer = buffer;
130  }
131  else if(m_depthBuffer == nullptr) {
132  m_depthBuffer = buffer;
133  }
134  m_buffers.push_back(buffer);
135  return buffer->renderbufferId;
136 }

169 void SFrameBuffer::RasterizeFramebuffer() {
170  glBindFramebuffer(GL_FRAMEBUFFER, m_fbo);
171  m_buffers.reserve(m_buffers.size());
172 
173  // tell OpenGL which color attachments we'll use (of this framebuffer) for rendering
174  if(m_colorAttachmentSize > 1) {
175  std::vector<unsigned int> attachments;
176  attachments.reserve(m_colorAttachmentSize);
177  for (int i = 0; i < m_colorAttachmentSize; ++i) {
178  attachments.push_back(GL_COLOR_ATTACHMENT0 + i);
179  }
180  glDrawBuffers(m_colorAttachmentSize, &attachments[0]);
181  }
182 
183  if(m_bufferStatus == COLOR_ONLY) {
184  GenerateTexturebuffer(SFrameBuffer::DEPTH, GL_DEPTH_COMPONENT);
185  m_bufferStatus = MULTI;
186  }
187 }

209 void SFrameBuffer::ResizeFrameBuffer(int width, int height) {
210  std::vector<BufferObject*> origBufferVector(m_buffers);
211 
212  m_buffers.clear();
213  m_buffers.reserve(origBufferVector.size());
214  m_colorAttachmentSize = 0;
215  glDeleteFramebuffers(1, &m_fbo);
216  GenerateFramebuffer(m_dimension, width, height);
217  m_depthBuffer = m_mainColorBuffer = nullptr;
218 
219  for (const auto& buffer : origBufferVector) {
220  auto type = buffer->type;
221  auto format = buffer->format;
222  auto level = buffer->level;
223  auto isTexture = buffer->texture != nullptr;
224 
225  ReleaseBufferObject(buffer);
226 
227  if(isTexture)
228  GenerateTexturebuffer(type, format, level);
229  else
230  GenerateRenderbuffer(type, format);
231  }
232  RasterizeFramebuffer();
233 }

235 void SFrameBuffer::BlitFrameBuffer(const SFrameBuffer& dst, BlitType type) {
236  if (m_mainColorBuffer == nullptr || m_depthBuffer == nullptr) {
237  Exterminate();
238  GenerateFramebuffer(PLANE, m_width, m_height);
239  GenerateTexturebuffer(RENDER, GL_RGB);
240  GenerateTexturebuffer(DEPTH, GL_DEPTH_COMPONENT);
241  RasterizeFramebuffer();
242  }
243 
244  const SFrameBuffer* a;
245  const SFrameBuffer* b;
246  if (type == REVERSE) {
247  a = this;
248  b = &dst;
249  }
250  else {
251  a = &dst;
252  b = this;
253  }
254  const auto& aColorTexture = a->m_mainColorBuffer->texture;
255  const auto& bColorTexture = b->m_mainColorBuffer->texture;
256  const auto& aDepthTexture = a->m_depthBuffer->texture;
257  const auto& bDepthTexture = b->m_depthBuffer->texture;
258 
259  if (m_blitObject.handle == nullptr) {
260  m_blitObject.handle = SResource::Create<GLProgramHandle>(Settings::GetDefaultBlitBufferShaderID());
261  m_blitObject.aColor = m_blitObject.handle->UniformLocation("a.color")->id;
262  m_blitObject.bColor = m_blitObject.handle->UniformLocation("b.color")->id;
263  m_blitObject.aDepth = m_blitObject.handle->UniformLocation("a.depth")->id;
264  m_blitObject.bDepth = m_blitObject.handle->UniformLocation("b.depth")->id;
265  }
266 
267  AttachFrameBuffer();
268  glViewport(0, 0, m_width, m_height);
269  glUseProgram(m_blitObject.handle->Program);
270  aColorTexture->Bind(m_blitObject.aColor, 0);
271  bColorTexture->Bind(m_blitObject.bColor, 1);
272  aDepthTexture->Bind(m_blitObject.aDepth, 2);
273  bDepthTexture->Bind(m_blitObject.bDepth, 3);
274 
275  ShaderUtil::BindAttributeToPlane();
276 }

286 int SFrameBuffer::GenerateAttachmentType(SFrameBuffer::BufferType type, bool isIncreaseAttachment) const {
287  switch (type) {
288  case RENDER:
289  if(m_bufferStatus == BufferStatus::NONE) m_bufferStatus = COLOR_ONLY;
290  if(m_bufferStatus == BufferStatus::DEPTH_ONLY) m_bufferStatus = MULTI;
291  if(isIncreaseAttachment)
292  return GL_COLOR_ATTACHMENT0 + (++m_colorAttachmentSize) - 1;
293  else
294  return GL_COLOR_ATTACHMENT0 + m_colorAttachmentSize;
295  case DEPTH:
296  if(m_bufferStatus == BufferStatus::NONE) m_bufferStatus = DEPTH_ONLY;
297  if(m_bufferStatus == BufferStatus::COLOR_ONLY) m_bufferStatus = MULTI;
298  return GL_DEPTH_ATTACHMENT;
299  case STENCIL:
300  default:
301  return GL_COLOR_ATTACHMENT0;
302  }
303 }
