43 unsigned int SFrameBuffer::GenerateRenderbuffer(BufferType type, int width, int height, int internalFormat) {

46  auto buffer = new BufferObject();

51  glGenRenderbuffers(1, &buffer->renderbufferId);

128 int SFrameBuffer::GenerateAttachmentType(SFrameBuffer::BufferType type, bool isIncreaseAttachment) const {
129  switch (type) {
130  case RENDER:
131  if(m_bufferStatus == BufferStatus::NONE) m_bufferStatus = COLOR_ONLY;
132  if(m_bufferStatus == BufferStatus::DEPTH_ONLY) m_bufferStatus = MULTI;
133  if(isIncreaseAttachment)
134  return GL_COLOR_ATTACHMENT0 + (m_colorAttachmentSize++);
135  else
136  return GL_COLOR_ATTACHMENT0 + m_colorAttachmentSize;
137  case DEPTH:
138  if(m_bufferStatus == BufferStatus::NONE) m_bufferStatus = DEPTH_ONLY;
139  if(m_bufferStatus == BufferStatus::COLOR_ONLY) m_bufferStatus = MULTI;
140  return GL_DEPTH_ATTACHMENT;
141  case STENCIL:
142  default:
143  return GL_COLOR_ATTACHMENT0;
144  }
145 }
