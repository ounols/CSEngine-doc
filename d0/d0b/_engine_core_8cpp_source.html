79void EngineCore::GenerateCores() {
80 if(m_isGenerated) return;
81 m_isGenerated = true;
82 m_cores = std::vector<CoreBase*>();
83 m_cores.reserve(ENGINE_COUNT);
84
85 m_resMgr = new ResMgr();
86 m_gameObjectMgr = new GameObjectMgr();
87 m_oglMgr = new OGLMgr();
88 m_renderMgr = new RenderMgr();
89 m_cameraMgr = new CameraMgr();
90 m_lightMgr = new LightMgr();
91 m_sceneMgr = new SceneMgr();
92 m_memoryMgr = new MemoryMgr();
93 m_scriptMgr = new ScriptMgr();
94
95 m_cores.push_back(m_resMgr);
96 m_cores.push_back(m_gameObjectMgr);
97 m_updateCores.push_back(m_gameObjectMgr);
98
99 m_cores.push_back(m_oglMgr);
100 m_renderCores.push_back(m_oglMgr);
101 m_cores.push_back(m_renderMgr);
102 m_renderCores.push_back(dynamic_cast<RenderCoreBase* const>(m_renderMgr));
103 m_cores.push_back(m_cameraMgr);
