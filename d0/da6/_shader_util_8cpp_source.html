23GLProgramHandle*
24ShaderUtil::CreateProgramHandle(const GLchar* vertexSource, const GLchar* fragmentSource, GLProgramHandle* handle) {
25 if (vertexSource == nullptr || fragmentSource == nullptr) return nullptr;
26 if (handle != nullptr && handle->Program != HANDLE_NULL) return nullptr;
27
28 GLProgramHandle* newHandle = handle;
29 auto program = createProgram(vertexSource, fragmentSource, *newHandle);
30 if (!program) {
31 return nullptr;
32 }
33
34 //Find important variables from shader.
35 auto variables_vert = GetImportantVariables(vertexSource);
36 auto variables_frag = GetImportantVariables(fragmentSource);
37
38 if (newHandle == nullptr)
39 newHandle = new GLProgramHandle();
40 newHandle->SetProgram(program);
41 //Get all variables from shader.
42 newHandle->SetAttributesList(variables_vert, variables_frag);
43 newHandle->SetUniformsList(variables_vert, variables_frag);
44
45 //Binding important variables to engine.
46 BindVariables(newHandle);
47
48 return newHandle;
49}

66GLuint ShaderUtil::createProgram(GLuint vertexShader, GLuint fragmentShader, const GLProgramHandle& handle) {
67 GLuint program = glCreateProgram();
68
69 if (program) {
70 //쉐이더를 attach 합니다.
71 glAttachShader(program, vertexShader);
72 glAttachShader(program, fragmentShader);
73 glLinkProgram(program);
74
75 GLint linkStatus = GL_FALSE;
76 glGetProgramiv(program, GL_LINK_STATUS, &linkStatus);
77 if (linkStatus != GL_TRUE) {
78 GLint bufLength = 0;
79 glGetProgramiv(program, GL_INFO_LOG_LENGTH, &bufLength);
80
81 if (bufLength) {
82 auto buf = static_cast<char*>(malloc(bufLength));
83
84 if (buf) {
85 glGetProgramInfoLog(program, bufLength, nullptr, buf);
86 auto errorLog = std::string("[") + handle.GetName() + "] Could not link program:" + buf;
87 SafeLog::Log(errorLog.c_str());
88 free(buf);
89
90 }
91 }
92 glDeleteProgram(program);
93 program = 0;
94
95 }
96 glDetachShader(program, vertexShader);
97 glDetachShader(program, fragmentShader);
98
99 }
100 glDeleteShader(vertexShader);
101 glDeleteShader(fragmentShader);
102
103 return program;
104}

106GLuint ShaderUtil::loadShader(GLenum shaderType, const char* pSource, const GLProgramHandle& handle) {
107 GLuint shader = glCreateShader(shaderType);
108 std::string srcString = m_defineVersion
109 + "#define MAX_JOINTS " + std::to_string(Settings::GetMaxJoints()) + '\n'
110 + "#define MAX_LIGHTS " + std::to_string(Settings::GetMaxLights()) + '\n'
111 + pSource;
112 const char* src = srcString.c_str();
113
114 if (shader) {
115 glShaderSource(shader, 1, &src, nullptr);
116 glCompileShader(shader);
117 GLint compiled = 0;
118 glGetShaderiv(shader, GL_COMPILE_STATUS, &compiled);
119
120 if (!compiled) {
121 GLint infoLen = 0;
122 glGetShaderiv(shader, GL_INFO_LOG_LENGTH, &infoLen);
123
124 if (infoLen) {
125 auto buf = static_cast<char*>(malloc(infoLen));
126
127 if (buf) {
128 glGetShaderInfoLog(shader, infoLen, nullptr, buf);
129 //LOGE("Could not compile shader %d:\n%s\n", shaderType, buf);
130 auto errorLog = std::string("[") + handle.GetName() + "] Could not compile shader:" + buf;
131 SafeLog::Log(errorLog.c_str());
132 free(buf);
133 }
134 glDeleteShader(shader);
135 shader = 0;
136 }
137 }
138 }
139 return shader;
140}

142std::map<std::string, std::string> ShaderUtil::GetImportantVariables(const GLchar* source) {
143 std::map<std::string, std::string> variables;
144 std::vector<std::string> str_line = split(source, ';');
145 std::string type_str;
146
147 for (auto line : str_line) {
148 line += ';';
149 auto result = split(line, '\n');
150
151 for (const auto& str : result) {
152 int start_index;
153 int end_index;
154 if ((start_index = str.find("//[")) != std::string::npos) {
155 start_index += 3;
156 end_index = str.find("]//");
157 type_str = str.substr(start_index, end_index - 3);
158 continue;
159 }
160
161 int eoc_index;
162 if (!type_str.empty() && (eoc_index = str.find(';')) != std::string::npos) {
163 int startIndex = str.substr(0, eoc_index).rfind(' ');
164 int endIndex = str.rfind('[');
165 endIndex = endIndex == std::string::npos ? eoc_index : endIndex;
166 auto detail = str.substr(startIndex, endIndex - startIndex);
167 detail = trim(detail);
168 variables[type_str] = detail;
169 type_str.clear();
170 }
171 }
172
173 }
174 return variables;
175}

246// position(x y z) + normal(x y z) + tex(u v) + weight(x y z) + jointID(id1, id2, id3)
247void ShaderUtil::BindAttributeToShader(const GLProgramHandle& handle, const GLMeshID& meshId) {
248 int stride = 4 * sizeof(vec3) + sizeof(vec2); //normal + position + uv = (4 * sizeof(vec3) + sizeof(vec2))
249 auto offset = (GLvoid*)sizeof(vec3);
250 GLint position = handle.Attributes.Position;
251 GLint normal = handle.Attributes.Normal;
252 GLint tex = handle.Attributes.TextureCoord;
253 GLint weight = handle.Attributes.Weight;
254 GLint jointId = handle.Attributes.JointId;
255
256 if(position < 0) return;
257
258 glBindBuffer(GL_ARRAY_BUFFER, meshId.m_vertexBuffer);
259 glEnableVertexAttribArray(position);
260 glVertexAttribPointer(position, 3, GL_FLOAT, GL_FALSE, stride, nullptr);
261 if(normal >= 0) {
262 glEnableVertexAttribArray(normal);
263 glVertexAttribPointer(normal, 3, GL_FLOAT, GL_FALSE, stride, offset);
264 }
265
266 offset = (GLvoid*)(sizeof(vec3) * 2);
267 if(tex >= 0) {
268 glEnableVertexAttribArray(tex);
269 glVertexAttribPointer(tex, 2, GL_FLOAT, GL_FALSE, stride, offset);
270 }
271
272 offset = (GLvoid*)(sizeof(vec3) * 2 + sizeof(vec2));
273 if(weight >= 0) {
274 glEnableVertexAttribArray(weight);
275 glVertexAttribPointer(weight, 3, GL_FLOAT, GL_FALSE, stride, offset);
276 }
277 offset = (GLvoid*)(sizeof(vec3) * 3 + sizeof(vec2));
278 if(jointId >= 0) {
279 glEnableVertexAttribArray(jointId);
280 glVertexAttribPointer(jointId, 3, GL_FLOAT, GL_FALSE, stride, offset);
281 }
282
283 if (meshId.m_indexSize < 0) {
284 glDrawArrays(GL_TRIANGLES, 0, meshId.m_vertexSize);
285 }
286 else {
287 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, meshId.m_indexBuffer);
288 glDrawElements(GL_TRIANGLES, meshId.m_indexSize * 3, GL_UNSIGNED_SHORT, nullptr);
289 }
290
291 glDisableVertexAttribArray(position);
292 if(normal >= 0) glDisableVertexAttribArray(normal);
293 if(tex >= 0) glDisableVertexAttribArray(tex);
294 if(weight >= 0) glDisableVertexAttribArray(weight);
295 if(jointId >= 0) glDisableVertexAttribArray(jointId);
296}

302void ShaderUtil::BindSkinningDataToShader(const GLProgramHandle& handle, const GLMeshID& meshId,
303 const std::vector<mat4>& jointMatrix) {
304 if (!meshId.m_hasJoint || jointMatrix.empty()) {
305 glUniform1i(handle.Uniforms.SkinningMode, 0);
306 return;
307 }
308
309 std::vector<float> result;
310 // result.reserve(jointMatrix.size() * 16 + 1);
311 for (const mat4& matrix : jointMatrix) {
312 const auto* pointer = matrix.Pointer();
313 for (int i = 0; i < 16; ++i, ++pointer) {
314 result.push_back(*(pointer));
315 }
316 }
317
318 glUniformMatrix4fv(handle.Uniforms.JointMatrix, Settings::GetMaxJoints(), GL_FALSE, &result[0]);
319 glUniform1i(handle.Uniforms.SkinningMode, 1);
320}
