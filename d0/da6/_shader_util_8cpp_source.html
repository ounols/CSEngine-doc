24 ShaderUtil::CreateProgramHandle(const GLchar* vertexSource, const GLchar* fragmentSource, GLProgramHandle* handle) {

44  gProgramhandle->SetUniformsList(variables_vert, variables_frag);

47  BindVariables(gProgramhandle);
48

106  }

161 std::map<std::string, std::string> ShaderUtil::GetImportantVariables(const GLchar* source) {
162  std::map<std::string, std::string> variables;
163 
164  std::vector<std::string> str_line = split(source, ';');
165 
166  std::string type_str = "";
167 
168  for (auto line : str_line) {
169  line += ';';
170  auto result = split(line, '\n');
171 
172  for (auto str : result) {
173  int start_index = 0;
174  int end_index = 0;
175  if ((start_index = str.find("//[")) != std::string::npos) {
176  start_index += 3;
177  end_index = str.find("]//");
178  type_str = str.substr(start_index, end_index - 3);
179  continue;
180  }
181 
182  int eoc_index = 0;
183  if (type_str != "" && (eoc_index = str.find(';')) != std::string::npos) {
184  int start_index = str.substr(0, eoc_index).rfind(' ');
185  int end_index = str.rfind('[');
186  end_index = end_index == std::string::npos ? eoc_index : end_index;
187  auto detail = str.substr(start_index, end_index - start_index);
188  detail = trim(detail);
189  variables[type_str] = detail;
190  type_str.clear();
191  }
192  }
193 
194  }
195 
196  return variables;
197 }

264 // position(x y z) + normal(x y z) + tex(u v) + weight(x y z) + jointID(id1, id2, id3)
265 void ShaderUtil::BindAttributeToShader(const GLProgramHandle& handle, const GLMeshID& meshId) {
266  int stride = 4 * sizeof(vec3) + sizeof(vec2); //normal + position + uv = (4 * sizeof(vec3) + sizeof(vec2))
267  const GLvoid* offset = (const GLvoid*) sizeof(vec3);
268  GLint position = handle.Attributes.Position;
269  GLint normal = handle.Attributes.Normal;
270  GLint tex = handle.Attributes.TextureCoord;
271  GLint weight = handle.Attributes.Weight;
272  GLint jointId = handle.Attributes.JointId;
273 
274  if (meshId.m_indexSize < 0) {
275  glBindBuffer(GL_ARRAY_BUFFER, meshId.m_vertexBuffer);
276  glVertexAttribPointer(position, 3, GL_FLOAT, GL_FALSE, stride, nullptr);
277  glVertexAttribPointer(normal, 3, GL_FLOAT, GL_FALSE, stride, offset);
278 
279  offset = (GLvoid*) (sizeof(vec3) * 2);
280  glVertexAttribPointer(tex, 2, GL_FLOAT, GL_FALSE, stride, offset);
281 
282  offset = (GLvoid*) (sizeof(vec3) * 2 + sizeof(vec2));
283  glVertexAttribPointer(weight, 3, GL_FLOAT, GL_FALSE, stride, offset);
284  offset = (GLvoid*) (sizeof(vec3) * 3 + sizeof(vec2));
285  glVertexAttribPointer(jointId, 3, GL_FLOAT, GL_FALSE, stride, offset);
286  glDrawArrays(GL_TRIANGLES, 0, meshId.m_vertexSize);
287  } else {
288  glBindBuffer(GL_ARRAY_BUFFER, meshId.m_vertexBuffer);
289  glVertexAttribPointer(position, 3, GL_FLOAT, GL_FALSE, stride, nullptr);
290  glVertexAttribPointer(normal, 3, GL_FLOAT, GL_FALSE, stride, offset);
291 
292  offset = (GLvoid*) (sizeof(vec3) * 2);
293  glVertexAttribPointer(tex, 2, GL_FLOAT, GL_FALSE, stride, offset);
294 
295  offset = (GLvoid*) (sizeof(vec3) * 2 + sizeof(vec2));
296  glVertexAttribPointer(weight, 3, GL_FLOAT, GL_FALSE, stride, offset);
297  offset = (GLvoid*) (sizeof(vec3) * 3 + sizeof(vec2));
298  glVertexAttribPointer(jointId, 3, GL_FLOAT, GL_FALSE, stride, offset);
299  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, meshId.m_indexBuffer);
300  glDrawElements(GL_TRIANGLES, meshId.m_indexSize * 3, GL_UNSIGNED_SHORT, 0);
301  }
302 }
