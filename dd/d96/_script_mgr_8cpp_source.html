87void ScriptMgr::RegisterScript(const std::string& script) {
88
89 HSQUIRRELVM vm = DefaultVM::Get();
90
91 //register script
92 if (!script.empty()) {
93 Script compiledScript;
94 compiledScript.CompileString(script);
95 if (Sqrat::Error::Occurred(vm)) {
96#ifdef WIN32
97 OutputDebugString(_SC("Compile Failed: "));
98 OutputDebugString(Error::Message(vm).c_str());
99#elif __ANDROID__
100 LOGE("Compile Failed : %s", Error::Message(vm).c_str());
101#elif __linux__
102 std::cout << "Compile Failed : " << Error::Message(vm) << '\n';
103#endif
104 }
105
106 compiledScript.Run();
107 if (Sqrat::Error::Occurred(vm)) {
108#ifdef WIN32
109 OutputDebugString(_SC("Run Failed: "));
110 OutputDebugString(Error::Message(vm).c_str());
111#elif __ANDROID__
112 LOGE("Run Failed : %s", Error::Message(vm).c_str());
113#elif __linux__
114 std::cout << "Run Failed : " << Error::Message(vm) << '\n';
115#endif
116 }
117
118 compiledScript.Release();
119 }

123SQInteger GetCustomComponentFunc(HSQUIRRELVM v) {
124 SQInteger args = sq_gettop(v);
125
126 if (args != 2) return 1;
127
128 SGameObject* game_object = Var<SGameObject*>(v, 1).value;
129 char* classname_str = Var<char*>(v, 2).value;
130 auto customComponent = game_object->GetCustomComponent(classname_str);
131
132 if (customComponent._type == OT_NULL) {
133 return 1;
134 }
135
136 sq_pushobject(v, customComponent);
137
138 return 1;
