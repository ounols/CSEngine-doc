39 bool STexture::Load(unsigned char* data) {
40 
41  if (m_texId != 0) {
42  stbi_image_free(data);
43  return false;
44  }
45 
46  glGenTextures(1, &m_texId);
47  glBindTexture(m_targetGL, m_texId);
48 
49  m_channels = GL_RGB;
50  if (m_channels == 4) m_channels = GL_RGBA;
51 
52  m_internalFormat = m_channels;
53  glTexImage2D(m_targetGL, 0, m_channels, m_width, m_height, 0, m_internalFormat, m_glType, data);
54 
55  glTexParameteri(m_targetGL, GL_TEXTURE_WRAP_S, GL_REPEAT);
56  glTexParameteri(m_targetGL, GL_TEXTURE_WRAP_T, GL_REPEAT);
57  glTexParameteri(m_targetGL, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
58  glTexParameteri(m_targetGL, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
59 
60  stbi_image_free(data);
61  return true;
62 }

116 bool STexture::InitTexture(int width, int height, int channel, int internalFormat, int glType) {
117  if (m_texId != 0) {
118  return false;
119  }
120 
121  m_width = width;
122  m_height = height;
123  m_channels = channel;
124  m_internalFormat = internalFormat;
125  m_glType = glType;
126 
127  glGenTextures(1, &m_texId);
128  glBindTexture(m_targetGL, m_texId);
129 
130  switch (m_type) {
131  case TEX_CUBE:
132  for (GLuint i = 0; i < 6; ++i) {
133  glTexImage2D(GL_TEXTURE_CUBE_MAP_POSITIVE_X + i, 0, m_internalFormat, m_width, m_height, 0, m_channels,
134  m_glType, nullptr);
135  }
136  break;
137 
138  default:
139  glTexImage2D(m_targetGL, 0, m_internalFormat, m_width, m_height, 0, m_channels, m_glType, nullptr);
140  }
141 
142  glTexParameteri(m_targetGL, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
143  glTexParameteri(m_targetGL, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
144  glTexParameteri(m_targetGL, GL_TEXTURE_WRAP_R, GL_CLAMP_TO_EDGE);
145  glTexParameteri(m_targetGL, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
146  glTexParameteri(m_targetGL, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
147 
148  return true;
149 }
