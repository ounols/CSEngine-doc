46 void DAEAnimationLoader::LoadAnimation() {
47  std::string rootNode = findRootJointName();
48  std::vector<float> times = getKeyTimes();
49  float duration = times[times.size() - 1];
50  std::vector<KeyFrameData*> keyFrames = initKeyFrames(times);
51  std::vector<XNode> animationNodes = m_animation.children;
52  for (const XNode& jointNode : animationNodes) {
53  if (jointNode.name != "animation") continue;
54  loadJointTransforms(keyFrames, jointNode, rootNode);
55  }
56 
57  m_animationData = new AnimationData(duration, keyFrames);
58 }

100 void DAEAnimationLoader::processTransforms(const std::string& jointName,
101  std::vector<float> rawData, std::vector<KeyFrameData*> keyFrames, bool root) {
102  for (int i = 0; i < keyFrames.size(); i++) {
103  std::vector<float> matrixData;
104  matrixData.reserve(16);
105  for (int j = 0; j < 16; j++) {
106  matrixData.push_back(rawData[i * 16 + j]);
107  }
108 
109  mat4 transform = mat4(&matrixData[0]);
110  transform = transform.Transposed();
111 
112  if (root) {
113  //because up axis in Blender is different to up axis in game
114  transform *= CORRECTION;
115  }
116 
117  keyFrames[i]->jointTransforms.push_back(
118  new JointTransformData(CORE->GetCore(ResMgr)->GetStringHash(jointName), jointName, transform));
119  }
120 }
