20void LightMgr::AttachLightToShader(const GLProgramHandle* handle) const {
21
22 if (handle == nullptr) return;
23
24// glUniform1i(handle->Uniforms.LightsSize, m_objects.size());
25
26 //std::vector<float> lightPosition;
27 //std::vector<int> lightType;
28 //std::vector<float> lightRadius;
29 //std::vector<float> lightColor;
30
31 int index = 0;
32 int shadow_index = 0;
33 for (const auto& light : m_objects) {
34
35 if (light == nullptr) continue;
36 if (!light->GetIsEnable()) continue;
37
38 LightComponent::LIGHT type = light->m_type;
39 SLight* lightObject = light->GetLight();
40
41 float lightPosition[4] = { 0.f };
42
43 //LightMode
44 //SetLightMode(handle, light, index);
45
46 switch (type) {
47
48 case LightComponent::DIRECTIONAL:
49 lightPosition[0] = lightObject->direction.x;
50 lightPosition[1] = lightObject->direction.y;
51 lightPosition[2] = lightObject->direction.z;
52 lightPosition[3] = lightObject->direction.w;
53 break;
54 case LightComponent::POINT:
55 lightPosition[0] = lightObject->position->x;
56 lightPosition[1] = lightObject->position->y;
57 lightPosition[2] = lightObject->position->z;
58 lightPosition[3] = 1.0f;
59 break;
60 case LightComponent::SPOT:
61 break;
62 case LightComponent::NONE:
63 default:
64 break;
65 }
66 glUniform4f(handle->Uniforms.LightPosition + index,
67 lightPosition[0], lightPosition[1], lightPosition[2], lightPosition[3]);
68 glUniform3f(handle->Uniforms.LightColor + index, lightObject->color.x, lightObject->color.y,
69 lightObject->color.z);
70 glUniform1i(handle->Uniforms.LightType + index, type);
71 if (handle->Uniforms.LightRadius >= 0)
72 glUniform1f(handle->Uniforms.LightRadius + index, lightObject->radius);
73 // Shadow
74 auto isShadow = light->IsShadow();
75 glUniform1i(handle->Uniforms.LightShadowMode + index, isShadow ? 1 : 0);
76 light->BindShadow(*handle, index, shadow_index);
77 if (isShadow) ++shadow_index;
78 ++index;
79 }
80
81 if (index <= 0) return;
82
83 glUniform1i(handle->Uniforms.LightSize, (int)m_objects.size());
84}
