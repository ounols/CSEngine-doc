64void AnimatorComponent::applyPoseToJoints(std::map<int, mat4>& currentPose, JointComponent* joint,
65 const mat4& parentTransform) {
66 const auto& object = joint->GetGameObject();
67 const int jointId = joint->GetAnimationJointId();
68 const mat4& currentLocalTransform = currentPose[jointId];
69 mat4&& currentTransform = currentLocalTransform * parentTransform;
70 auto children = object->GetChildren();
71 for (const auto& child : children) {
72 const auto& joint_component = child->GetComponent<JointComponent>();
73 if (joint_component != nullptr)
74 applyPoseToJoints(currentPose, joint_component, currentTransform);
75 }
76 currentTransform = joint->GetInverseTransformMatrix() * currentTransform;
77 joint->SetAnimationMatrix(std::move(currentTransform));
78}

80std::vector<KeyFrame*> AnimatorComponent::getPreviousAndNextFrames() const {
81 auto allFrames = m_currentAnimation->GetKeyFrames();
82 KeyFrame* previousFrame = allFrames.front();
83 KeyFrame* nextFrame = allFrames.front();
84 allFrames.pop_front();
85
86 for (const auto& frame : allFrames) {
87 nextFrame = frame;
88 if (nextFrame->GetTimeStamp() > m_animationTime) {
89 break;
90 }
91 previousFrame = frame;
92 }
93
94 std::vector<KeyFrame*> result;
95 result.reserve(2);
96 result.push_back(previousFrame);
97 result.push_back(nextFrame);
98 return result;
99}

101float AnimatorComponent::CalculateProgression(KeyFrame* previous, KeyFrame* next) const {
102 float totalTime = next->GetTimeStamp() - previous->GetTimeStamp();

107std::map<int, mat4> AnimatorComponent::InterpolatePoses(KeyFrame* previousFrame, KeyFrame* nextFrame, float t) {
108 std::map<int, mat4> currentPose;
109 auto jointKeyFrames_prev = previousFrame->GetJointKeyFrames();
110 auto jointKeyFrames_next = nextFrame->GetJointKeyFrames();
111 for (const auto& frame : jointKeyFrames_prev) {
112 const auto jointId = frame.first;
113 JointTransform* previousTransform = frame.second;
114 JointTransform* nextTransform = jointKeyFrames_next[jointId];
115 JointTransform currentTransform = JointTransform::Interpolate(t, *previousTransform, *nextTransform);
116 currentPose.insert(std::pair<int, mat4>(jointId, currentTransform.GetLocalMatrix()));
117 }
118 return currentPose;
119}
