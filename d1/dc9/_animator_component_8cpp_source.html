109 std::map<int, mat4> AnimatorComponent::InterpolatePoses(KeyFrame* previousFrame, KeyFrame* nextFrame, float t) {
110  std::map<int, mat4> currentPose;
111  auto jointKeyFrames_prev = previousFrame->GetJointKeyFrames();
112  auto jointKeyFrames_next = nextFrame->GetJointKeyFrames();
113  for (const auto& frame : jointKeyFrames_prev) {
114  const auto jointId = frame.first;
115  JointTransform* previousTransform = frame.second;
116  JointTransform* nextTransform = jointKeyFrames_next[jointId];
117  JointTransform currentTransform = JointTransform::Interpolate(t, *previousTransform, *nextTransform);
118  currentPose.insert(std::pair<int, mat4>(jointId, currentTransform.GetLocalMatrix()));
119  }
120  return currentPose;
121 }
