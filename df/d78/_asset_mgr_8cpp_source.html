61AssetMgr::AssetReference* AssetMgr::GetAsset(std::string name) const {
62 make_lower(name);
63 for (auto asset : m_assets) {
64 if (make_lower_copy(asset->name) == name) return asset;
65 if (make_lower_copy(asset->id) == name) return asset;
66 if (make_lower_copy(asset->path) == name) return asset;
67 if (make_lower_copy(asset->name_full) == name) return asset;
68 }
69 auto errorLog = "[Assets Warning] " + name + " does not exist.";
70 SafeLog::Log(errorLog.c_str());

74void AssetMgr::ReadDirectory(const std::string& path) {
75#ifdef __ANDROID__
76 return;
77 AAssetDir* assetDir = AAssetManager_openDir(m_assetManager, "");
78 const char* filename = (const char*)NULL;
79 SafeLog::Log("ReadDirectory");
80 while ((filename = AAssetDir_getNextFileName(assetDir)) != NULL) {
81 AAsset* asset = AAssetManager_open(m_assetManager, filename, AASSET_MODE_STREAMING);
82
83 std::string name = filename;
84 SafeLog::Log(name.c_str());
85 AssetReference* assetReference = CreateAsset(path + name, name);
86
87 AAsset_close(asset);
88 }
89 AAssetDir_close(assetDir);
90#elif __linux__ //======================================
91
92 DIR* dirp = opendir(path.c_str());
93 struct dirent* dp;
94 while ((dp = readdir(dirp)) != nullptr) {
95
96 std::string name = dp->d_name;
97
98 if (dp->d_type == DT_DIR) {
99 if (name == "." || name == "..") continue;
100
101 ReadDirectory(path + name + '/');
102 continue;
103 }
104
105 AssetReference* asset = CreateAsset(path + name, name);
106 std::cout << "[pkg] " << asset->name << " (" << asset->extension << ")\n";
107 }
108 closedir(dirp);
109
110 return;
111
112#endif //================================================
113#ifdef _WIN32 //==========================================
114 WIN32_FIND_DATA data;
115 HANDLE hFind = FindFirstFile(std::string(path + '*').c_str(), &data);
116 if (hFind != INVALID_HANDLE_VALUE) {
117 do {
118 std::string name = data.cFileName;
119
120 if (data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) {
121 if (name == "." || name == "..") continue;
122
123 ReadDirectory(path + name + '/');
124 continue;
125 }
126
127 AssetReference* asset = CreateAsset(path + name, name);
128 std::cout << "[pkg] " << asset->name << " (" << asset->extension << ")\n";
129
130 } while (FindNextFile(hFind, &data) != 0);
131 FindClose(hFind);
132 }
133#endif //================================================
134}

136void AssetMgr::ReadPackage(const std::string& path) {
137#ifdef __ANDROID__
138 if(m_zip == nullptr) {
139 SafeLog::Log(path.c_str());
140 m_package_raw = OpenNativeAssetsTxtFile(path);
141 m_zip = zip_stream_open(m_package_raw.c_str(), m_package_raw.length(), 0, 'r');
142 }
143#else
144 if(m_zip == nullptr) m_zip = zip_open(path.c_str(), 0, 'r');
145#endif
146 int zipSize = zip_entries_total(m_zip);
147 for (int i = 0; i < zipSize; ++i) {
148 zip_entry_openbyindex(m_zip, i);
149 {
150 const char *name = zip_entry_name(m_zip);
151 std::string name_str = name;
152 auto rFindIndex = name_str.rfind('/');
153 std::string name_cropped = (rFindIndex == std::string::npos) ? name_str : name_str.substr(rFindIndex + 1);
154 int isdir = zip_entry_isdir(m_zip);
155 if(isdir != 0) continue;
156 AssetReference* asset = CreateAsset(name, name_cropped);
157 std::cout << "(Packed)[pkg] " << asset->name << " (" << asset->extension << ")\n";
158 }
159 zip_entry_close(m_zip);
160 }

164AssetMgr::AssetReference* AssetMgr::CreateAsset(const std::string& path, const std::string& name_full, std::string name) {
165 auto asset = new AssetReference();
166 asset->path = path;
167 asset->id = "File:" + path.substr(CSE::AssetsPath().size());
168 asset->name_full = name_full;
169
170 if (name.empty()) {
171 name = name_full;
172 auto name_strs = split(name, '.');
173 asset->extension = name_strs[name_strs.size() - 1];
174 asset->name = name.substr(0, name.rfind('.'));
175 } else {
176 asset->name = name;
177 }
178
179
180 m_assets.push_back(asset);
181
182 return asset;
183}

185void AssetMgr::SetType() {
186
187 for (int i = 0; i < m_assets.size(); ++i) {
188 const auto asset = m_assets[i];
189 std::string type_str = asset->extension;
190 make_lower(type_str);
191
192 //차후 확장자 리스트를 제작하여 최적화
193 //texture data
194 if (type_str == "jpg" || type_str == "png" || type_str == "dds" || type_str == "hdr") {
195 asset->type = TEX_2D;
196 asset->name += ".texture";
197 continue;
198 }
199
200 //cube map texture data
201 if (type_str == "cbmap") {
202 asset->type = TEX_CUBEMAP;
203 asset->name += ".textureCubeMap";
204 continue;
205 }
206
207 // framebuffer data
208 if (type_str == "framebuffer") {
209 asset->type = FRAMEBUFFER;
210 asset->name += ".frameBuffer";
211 continue;
212 }
213
214 //material data
215 if (type_str == "mat") {
216 asset->type = MATERIAL;
217 asset->name += ".material";
218 continue;
219 }
220
221 //DAE data
222 if (type_str == "dae") {
223 asset->type = DAE;
224 asset->name += ".prefab";
225 {
226 AppendSubName(CreateAsset(asset->path, asset->name_full, asset->name), "animation")->type = DAE;
227 AppendSubName(CreateAsset(asset->path, asset->name_full, asset->name), "skeleton")->type = DAE;
228 AppendSubName(CreateAsset(asset->path, asset->name_full, asset->name), "mesh")->type = DAE;
229 }
230 continue;
231 }
232
233 //script data
234 if (type_str == "nut") {
235 asset->type = SCRIPT;
236 asset->name += ".script";
237 continue;
238 }
239
240 //shader part data
241 if (type_str == "vert" || type_str == "vs") {
242 asset->type = SHADER;
243 asset->name += ".vert";
244 continue;
245 }
246 if (type_str == "frag" || type_str == "fs") {
247 asset->type = SHADER;
248 asset->name += ".frag";
249 continue;
250 }
251
252 //shader part data
253 if (type_str == "shader") {
254 asset->type = SHADER_HANDLE;
255 asset->name += ".shader";
256 continue;
257 }
258
259 //prefab data
260 if (type_str == "prefab") {
261 asset->type = PREFAB;
262 asset->name += ".prefab";
263 continue;
264 }
265
266 //scene data
267 if (type_str == "scene") {
268 asset->type = SCENE;
269 asset->name += ".scene";
270 continue;
271 }
272
273 //ini data
274 if (type_str == "ini") {
275 asset->type = INI;
276 asset->name += ".ini";
277 continue;
278 }
279
280 if(asset->type == NONE) {
281 //txt data
282 asset->type = TXT;
283 asset->name += ".text";
284 }
285
286 }

309std::string AssetMgr::LoadAssetFile(const std::string& path) {
310 if(Settings::IsAssetsPacked() == false)
311 return OpenNativeAssetsTxtFile(path);
312
313 std::string result;
314 const auto& zip = CORE->GetResMgrCore()->m_assetManager->m_zip;
315 zip_entry_open(zip, path.c_str());
316 {
317 char* buf = nullptr;
318 size_t bufsize = 0;
319
320 zip_entry_read(zip, (void **)&buf, &bufsize);
321 result = std::string(buf, bufsize);
322 free(buf);
323 }
324 zip_entry_close(zip);
325 return result;
326}
