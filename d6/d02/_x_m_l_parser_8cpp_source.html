63STexture* XMLParser::parseTexture(const char* value) {
64 const auto& asset = CORE->GetCore(ResMgr)->GetAssetReference(value);
65 if(asset == nullptr || asset->type == AssetMgr::FRAMEBUFFER) {
66 const auto value_split = split(value, '?');
67 if(value_split.size() <= 1)
68 return SResource::Create<STexture>(value);
69
70 const auto& frameBuffer = SResource::Create<SFrameBuffer>(value_split[0]);
71 return frameBuffer->GetTexture(value);
72 }
73 return SResource::Create<STexture>(value);
74}

76SFrameBuffer* XMLParser::parseFrameBuffer(const char* value) {
77 std::string value_result = value;
78 const auto split_index = value_result.rfind('?');
79 if(split_index != std::string::npos)
80 value_result = value_result.substr(0, split_index - 1);
81
82 const auto& asset = CORE->GetCore(ResMgr)->GetAssetReference(value_result);
83 if(asset->type != AssetMgr::FRAMEBUFFER) {
