73void SPrefab::GenerateResourceID(SGameObject* obj) {
74
75 if (obj == nullptr) {
76 obj = m_root;
77 obj->SetResourceID(GetID());
78 } else {
79 std::string resultID = obj->GetID();
80 resultID = resultID.substr(resultID.find('/') + 1);
81 obj->SetResourceID(std::string(GetID()) + "*" + resultID);
82 }
83
84 for (auto child : obj->GetChildren()) {
85 GenerateResourceID(child);
86 }

88}
