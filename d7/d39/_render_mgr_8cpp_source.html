43void RenderMgr::Render() const {
44 // Render Order : Depth Buffers -> Sub Render Buffers -> Main Render Buffers
45 // 1. Render depth buffer for shadows.
46 const auto& lightObjects = lightMgr->GetAll();
47 const auto& shadowObjects = lightMgr->GetShadowObject();
48 const auto& shadowHandle = lightMgr->GetShadowHandle();
49 for (const auto& light : lightObjects) {
50 if(!light->IsShadow()) continue;
51 RenderShadowInstance(*light, *shadowHandle, shadowObjects);
52 }
53 lightMgr->RefreshShadowCount();
54 const auto& cameraObjects = cameraMgr->GetAll();
55 const auto& mainCamera = cameraMgr->GetCurrentCamera();
56
57 // 2. Render active sub cameras.
58 for (const auto& camera : cameraObjects) {
59 if(!camera->GetIsEnable() || camera == mainCamera || camera->GetFrameBuffer() == nullptr) continue;
60 ResetBuffer(*camera);
61 RenderGbuffers(*camera); // Deferred Render
62 RenderInstances(*camera); // Forward Render
63 }
64 if(mainCamera == nullptr) return;
65
66 // 3. Main Render Buffer
67 ResetBuffer(*mainCamera);
68 RenderGbuffers(*mainCamera); // Deferred Render
69 RenderInstances(*mainCamera); // Forward Render
70

163void RenderMgr::RenderInstances(const CameraBase& camera, const GLProgramHandle* custom_handler) const {
164 const auto cameraMatrix = camera.GetCameraMatrixStruct();
165 const auto& frameBuffer = camera.GetFrameBuffer();
166 int customHandlerID = custom_handler != nullptr ? (int)custom_handler->Program : -1;
167 OrderRenderLayer orderRenderLayer(m_rendersLayer.begin(), m_rendersLayer.end());
168 if(frameBuffer == nullptr) {
169 GetMainBuffer()->AttachFrameBuffer();
170 }
171 else {
172 // If the framebuffer is a depth buffer
173 if(frameBuffer->GetBufferStatus() == SFrameBuffer::DEPTH_ONLY) {
174 customHandlerID = (int)lightMgr->GetShadowHandle()->Program;
175 }
176 frameBuffer->AttachFrameBuffer();
177 }
178
179 for (const auto& orderLayerPair : orderRenderLayer) {
180 const auto& orderLayer = orderLayerPair.second;
181 ProgramRenderLayer programComp(orderLayer.begin(), orderLayer.end());
182
183 for (const auto& programPair : programComp) {
184 const auto& handler = *programPair.first;
185 const auto& renderComp = programPair.second;
186
187 if (programPair.first == nullptr) continue;
188 if (renderComp.empty()) continue;
189
190 glUseProgram(customHandlerID < 0 ? handler.Program : customHandlerID);
191 //Attach Light
192 lightMgr->AttachLightToShader(&handler);
193
194 for (const auto& render : renderComp) {
195 if (render == nullptr) continue;
196 if (!render->isRenderActive) continue;
197
198 render->SetMatrix(cameraMatrix);
199 render->Render();
200 }
201 }
202 }
203
205 glBindBuffer(GL_ARRAY_BUFFER, 0);
206 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
207
208}

210void RenderMgr::RenderShadowInstance(const CameraBase& camera, const GLProgramHandle& custom_handler,
211 const std::list<SIRender*>& render_objects) const {
212 const auto cameraMatrix = camera.GetCameraMatrixStruct();
213 const auto& frameBuffer = camera.GetFrameBuffer();
214 int customHandlerID = (int)custom_handler.Program;
215 if(frameBuffer == nullptr) {
216 GetMainBuffer()->AttachFrameBuffer();
217 }
218 else {
219 // If the framebuffer is a depth buffer
220 if(frameBuffer->GetBufferStatus() == SFrameBuffer::DEPTH_ONLY) {
221 customHandlerID = (int)lightMgr->GetShadowHandle()->Program;
222 }
223 frameBuffer->AttachFrameBuffer();
224 }
225
226 glUseProgram(customHandlerID);
227 // Initialize various state.
228
229 glClear(GL_DEPTH_BUFFER_BIT);
230 for (const auto& shadowObject : render_objects) {
231 if(!shadowObject->isRenderActive) continue;
232 const auto& shadow_transform = static_cast<RenderComponent*>(shadowObject)->GetGameObject()->GetTransform();
233
234 if(LightMgr::SHADOW_DISTANCE < vec3::Distance(cameraMatrix.cameraPosition, shadow_transform->m_position))
235 continue;
236
237 shadowObject->SetMatrix(cameraMatrix, &custom_handler);
238 shadowObject->Render(&custom_handler);
239 }
240 glBindFramebuffer(GL_FRAMEBUFFER, 0);
241}
